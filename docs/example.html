Quicksort Algorithm

import numpy as np

def quicksort(array):
    """Sort an array using the quick sort algorithm.
    Takes an array as argument.
    This implementation is recursive."""

    if len(array) <= 1:
        return array

    # Get the element in the center of the array
    middle_point = int(len(array) / 2)

    # Calculate the median of 3 values, the first,
    # the middle and the last elements of the array
    pivot = np.median([array[0], array[middle_point], array[len(array)-1]])

    # Filter array values by its relation with the pivot
    smaller = [x for x in array if x < pivot]
    equal = [x for x in array if x== pivot]
    larger = [x for x in array if x > pivot]

    return quicksort(smaller) + equal + quicksort(larger)

print(quicksort([5, 4, 3, 2, 1]))

(def polygon
  [
   [-3 -2]
   [-1 4]

(defn cross-multiplication [points pa pb]
  (map #(* (first %) (second %))
       (map (fn [col-a col-b] [(pa col-a) (pb col-b)])
            points (rest points))))

(defn polygon-area [pol]
  (/ (- (reduce + (cross-multiplication pol first second))
        (reduce + (cross-multiplication pol second first)))
     2))

(polygon-area polygon)
;; => 60

(defrecord Coordinate [x y])
(defrecord Polygon [points])

(defn map-points
  "Cross the points `pa` `pb` from a polygon and map them together.
  It returns a map of [(pa point-n) (pb point-n1)]."
  [pol pa pb]
  (let [points (:points pol) ; Get points from polygon
        ;; Remove first element and append it at the end to cross
        ;; the points
        cross (conj (into [] (rest points)) (first points))]
    (map
     (fn [a b]
       [(pa a) (pb b)])
     points cross)))

(defn cross-product [mapped-points]
  (map #(* (first %) (second %)) mapped-points))

(defn area-of-polygon [pol]
  (/ (- (reduce + (cross-product (map-points pol :x :y)))
        (reduce + (cross-product (map-points pol :y :x))))
     2))

(def irregular-polygon
  "Define a new irregular polygon with 5 points."
  (->Polygon [
              (->Coordinate -3 -2)
              (->Coordinate -1 4)
              (->Coordinate 6 1)
              (->Coordinate 3 10)
              (->Coordinate -4 9)
              ]))

(area-of-polygon irregular-polygon)
;; => 60

    class Net(nn.Module):
    def __init__(self, input_dim, units, output_dim):
        """Define all the layers on the neural network."""
        super(Net, self).__init__()

        # Input Layers:
        # - feed forward neural network
        # - lstm
        self.linear_1 = nn.Linear(input_dim, units)
        self.lstm_1 = nn.LSTM(input_dim, units, num_layers=2)
        
        # Hidden Layers
        self.linear_2 = nn.Linear(units, units)
        self.linear_3 = nn.Linear(units, units)
        
        # This layer outputs the LSTM layer and has the same shape as the
        # second layer's input data.
        self.linear_4 = nn.Linear(units, TIMESTEPS * 7) # LSTM network
        self.relu_1 = nn.ReLU()
        self.relu_2 = nn.ReLU()
        self.relu_3 = nn.ReLU()
        self.relu_4 = nn.ReLU()
        self.drop_1 = nn.Dropout(p=0.5)
        self.drop_2 = nn.Dropout(p=0.5)
        
        # Output Layers
        self.linear_output_1 = nn.Linear(units, output_dim)
        self.linear_output_2 = nn.Linear(units, output_dim)
        self.linear_output_3 = nn.Linear(units, output_dim)
        self.linear_output_4 = nn.Linear(units, output_dim)
        self.linear_output_5 = nn.Linear(units, output_dim)
        self.linear_output_6 = nn.Linear(units, output_dim)
        self.linear_output_7 = nn.Linear(units, output_dim)
        
        # Output activation
        self.relu_out_1 = nn.ReLU()
        self.relu_out_2 = nn.ReLU()
        self.relu_out_3 = nn.ReLU()
        self.relu_out_4 = nn.ReLU()
        self.relu_out_5 = nn.ReLU()
        self.relu_out_6 = nn.ReLU()
        self.relu_out_7 = nn.ReLU()
        
    def forward(self, x, y):
        """Define how the layers interact with the data."""
        
        # LSTM Network
        lstm_out, lstm_hidden = self.lstm_1(y)
        y = self.relu_4(lstm_out)
        y = self.linear_4(y).view(-1, TIMESTEPS * 7)
        
        # Join output of LSTM and Feed Forward Network input
        x = torch.cat((x, y))
        
        # Feed Forward Network
        x = self.linear_1(x)
        x = self.relu_1(x)
        x = self.drop_1(x)
        
        x = self.linear_2(x)
        x = self.relu_2(x)
        x = self.drop_2(x)
        
        x = self.linear_3(x)
        x = self.relu_3(x)
        
        # Outputs
        output_1 = self.linear_output_1(x)
        output_1 = self.relu_out_1(output_1)
        
        output_2 = self.linear_output_2(x)
        output_2 = self.relu_out_2(output_2)
        
        output_3 = self.linear_output_3(x)
        output_3 = self.relu_out_3(output_3)
        
        output_4 = self.linear_output_4(x)
        output_4 = self.relu_out_4(output_4)
        
        output_5 = self.linear_output_5(x)
        output_5 = self.relu_out_5(output_5)
        
        output_6 = self.linear_output_6(x)
        output_6 = self.relu_out_6(output_6)
        
        output_7 = self.linear_output_7(x)
        output_7 = self.relu_out_7(output_7)
        
        return output_1, output_2, output_3, output_4, output_5, output_6, output_7
